# create namespace
CREATE KEYSPACE company WITH replication = {'class':'SimpleStrategy','replication_factor':1};
DESCRIBE KEYSPACES;

USE company;

# create table
CREATE TABLE employees ( employee_id UUID PRIMARY KEY, first_name TEXT, last_name TEXT, position TEXT, 
salary DOUBLE, date_of_joining DATE );
DESCRIBE TABLES;
